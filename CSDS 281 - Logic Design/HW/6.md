## alu_parm.sv
```systemverilog
module alu_parm #(
    parameter N = 4
) (
    input logic [N-1:0] a, b,
    input logic c_in,
    output logic [N-1:0] s,
    output logic c_out
);
    logic [N-1:0] p, g;
    logic [N:0] c;

    always_comb begin
        p = a ^ b;
        g = a & b;

        c[0] = c_in;
        for (int i = 0 ; i < N; i++) begin
            c[i+1] = g[i] | (p[i] & c[i]);
        end
        s = p ^ c;
        c_out = c[N];
    end
endmodule
```
